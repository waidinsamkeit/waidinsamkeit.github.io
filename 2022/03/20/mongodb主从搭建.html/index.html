Mongodb主从搭建

  • 内存2以上
  • 无特殊要求
  • 主IP：192.168.1.100
  • 从IP：192.168.1.101
  • 准备配置如下，每台服务器都执行
1sudo echo "never" > /sys/kernel/mm/transparent_hugepage/enabled
2sudo echo "never" >  /sys/kernel/mm/transparent_hugepage/defrag
3vim /etc/security/limits.conf
4# 添加mongo用户可以打开的文件数量的限制
5mongod  soft  nofile  64000
6mongod  hard  nofile  64000
7mongod  soft  nproc  32000
8mongod  hard  nproc  32000

1.0 下载安装Mongo

1[root@bogon ~]# curl -O http://fastdl.mongodb.org/linux/mongodb-linux-x86_64-3.4.2.tar.gz

1[root@bogon ~]# tar -zxf mongodb-linux-x86_64-3.4.2.tgz # 解压 
2[root@bogon ~]# mv mongodb-linux-x86_64-3.4.2 /usr/local/mongo # 移动目录到/usr/local/mongo

1.2 主mongo配置

 1[root@bogon mongo]# mkdir /usr/local/mongo/{conf,data,logs}
 2[root@bogon mongo]# vim /usr/local/mongo/conf/mongo.conf
 3port=27017
 4fork=true
 5logpath=/usr/local/mongo/logs/mongodb.log
 6logappend=true
 7dbpath=/usr/local/mongo/data
 8maxConns=1024
 9master=true
10oplogSize=2048

1port=27017       #端口号
2fork=true        #以守护进程方式运行
3logpath=/usr/local/mongodb/logs/mongodb.log  #日志文件
4logappend=true   #日志输出方式
5dbpath=/usr/local/mongodb/data  #数据库位置
6maxConns=1024    #数据库最大连接数
7master=true      #主模式
8oplogSize=2048   #日志滚动，单位M

2fork=true
3logpath=/usr/local/mongo/logs/mongodb.log
4logappend=true

1[root@localhost ~]# ln -s /usr/local/mongo/bin/* /usr/bin/

1[root@bogon mongo]# mongod -f /usr/local/mongo/conf/mongo.conf 
2# -f 指定配置文件
3[root@bogon mongo]# mongod -f /usr/local/mongo/conf/mongo.conf --shutdown //关闭mongo

问题解决

  • WARNING: Access control is not enabled for the database

原因分析：新版本的MongDB增加了安全性设计，推荐用户创建使用数据库时进行验证。如果用户想建立简单连接，则会提示警示信息。

解决方法

1[root@bogon logs]# mongo # 进入mongo数据库
2use admin //进入admin
3db.createUser({user: 'root', pwd: '123456qwerty!@#$%^', roles: ['root']}) # 创建用户密码和权限
4db.auth('root', '123456qwerty!@#$%^') # 返回1 认证成功，返回0认证失败
5# 因为mongo的密码是针对库设置的，不是像mysql一样针对全局设置的
  • “errmsg” : “not master and slaveOk=false”

如果从服务器上进入mongo以后使用show dbs查看是否同步数据库报这个错误是正常的

关于Mongo认证机制

  • --auth:表示带着认证方式进行启动
1mongod -f /usr/local/mongo/conf/mongo.conf --auth

进入到mongo使用show dbs发现报错

1not authorized on admin to execute command { listDatabases: 1.0 }

此时应该先到admin库中进行验证

1> use admin
2> db.auth('root', '123456qwerty!@#$%^')  # 返回结果为1则认证成功
3> show dbs # 此时查看数据就可以了
