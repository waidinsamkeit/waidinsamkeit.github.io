脚本每个人的环境不同.可能有的会有问题，按照自己的环境来改和执行

 1[root@localhost redis]# vim start.sh  //就是不想手动敲那么累，全是命令拼凑
 2#!/bin/bash
 3Redis_home=/usr/local/redis
 4# start dow redis
 5echo -e "\033[41;36m test env  \033[0m"
 6if [ ! -d "/backup" ]; then
 7  mkdir /backup
 8fi
 9mv /etc/yum.repos.d/* /backup
10wget -O /etc/yum.repos.d/CentOS-Base.repo http://mirrors.aliyun.com/repo/Centos-7.repo
11wget -O /etc/yum.repos.d/epel.repo http://mirrors.aliyun.com/repo/epel-7.repo
12# installk envermint
13echo -e "\033[41;36m install env \033[0m"
14yum makecache > /dev/null 2>&1
15yum -y install centos-release-scl  scl-utils-build
16yum -y install devtoolset-7-gcc* gcc
17scl enable devtoolset-7 bash
18# zip redis
19echo -e "\033[41;36m install redis  \033[0m" 
20wget http://download.redis.io/releases/redis-6.0.8.tar.gz
21tar -zxf redis-6.0.8.tar.gz
22mv  redis-6.0.8 $Redis_home
23cd /usr/local/redis && make install

1.0 修改主Redis配置文件

1[root@localhost redis]# vim /usr/local/redis/redis.conf
2bind 192.168.1.100  # 改为本机IP
3port 6379   # 默认6379
4daemonize yes # 改为yes允许后台执行
5pidfile /var/run/redis_6379.pid #  pid文件位置
6maxmemory 750mb # 设置最大内存 一般为3/4
7dbfilename redis-master.rdb #  改一下rdb文件名称
8logfile "/usr/local/redis/logs/redis-master.log" # 日志文件
9requirepass 123.com  # redis密码

1.1 修改从Redis配置文件

 1[root@localhost ~]# vim /usr/local/redis/redis.conf
 2bind 192.168.1.101  # 改为本机IP
 3port 6379   # 默认6379
 4daemonize yes  # 改为yes允许后台执行
 5pidfile /var/run/redis_6379.pid #  pid文件位置
 6maxmemory 750mb # 设置最大内存 一般为3/4
 7dbfilename redis-slave.rdb # 改一下rdb文件名称
 8replicaof 192.168.1.100 6379 # 从节点跟随主节点的地址
 9logfile "/usr/local/redis/logs/redis-slave.log" # 出去记得自己创建目录
10requirepass 123.com

1.3 启动Redis主从

1echo "511" > /proc/sys/net/core/somaxconn
2echo "vm.overcommit_memory = 1" >> /etc/sysctl.conf
3sysctl -p 
4[root@localhost ~]# cd /usr/local/redis/;redis-server redis.conf

检测主从是否同步

1[root@localhost redis]# redis-cli -c -h 192.168.1.100 -p 6379 -a 123.com  # 从上面链接主
2192.168.1.100:6379> set for ceshi  
3[root@localhost redis]# redis-cli -c -h 192.168.1.101 -p 6379 -a 123.com # 在切换到从服务器
4192.168.1.101:6379> get for
5"ceshi"  # 同步到数据即可
