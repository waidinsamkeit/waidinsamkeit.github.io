1.0 准备条件

准备如下：

  • filebeat-6.3.2
  • elasticsearch-6.3.2
  • kibana-6.3.2
  • logstash-6.3.2
  • 内存3GB以上,elasticsearch这东西太吃内存了,实验环境建议内存给到4G.
  • 官网地址 https://www.elastic.co/cn/elasticsearch/

如果你嫌弃官网下载的太慢,可以使用以下微云地址下载

  • https://share.weiyun.com/fnAz5I2f
  • 版本最好是和我这个一致吧

教程是老教程了,微云这会儿也限速了 - 2022-03-21

2.0 搭建Elasticsearch

2.0.1 准备JDK环境

  • 下载JDK8
 1#  解压并且移动jdk目录
 2[root@MySQL8 ~]# tar -zxf jdk-8u201-linux-x64.tar.gz
 3[root@MySQL8 ~]# mv jdk1.8.0_201/ /usr/local/java
 4#  设置java的环境变量
 5echo '
 6export JAVA_HOME=/usr/local/java
 7export JRE_HOME=/usr/local/java/jre
 8export CLASSPATH=$JAVA_HOME/lib:$JRE_HOME/lib
 9export PATH=$PATH:$JAVA_HOME/bin:$JRE_HOME/bin
10' >> /etc/profile
11[root@MySQL8 ~]# source /etc/profile  # 加载profile使其生效
12# 检测配置的java环境变量是否生效
13[root@MySQL8 ~]# java -version # 使用java -verison出现版本信息
14java version "1.8.0_201"
15Java(TM) SE Runtime Environment (build 1.8.0_201-b09)
16Java HotSpot(TM) 64-Bit Server VM (build 25.201-b09, mixed mode)

2.0.2 部署Elasticsearch单机版

1[root@localhost ~]# tar -zxf elasticsearch-6.3.2.tar.gz
2[root@localhost ~]# mv elasticsearch-6.3.2 /usr/local/elasticsearch-6.3.2
3[root@localhost ~]# cd /usr/local/elasticsearch-6.3.2/config/
4[root@localhost ~]# vim elasticsearch.yml

修改以下内容

 1# 有注释的把注释取消,没有注释的手动添加
 2# ---------------------------------- Cluster -----------------------------------
 3cluster.name: my-application
 4# ------------------------------------ Node ------------------------------------
 5node.name: node-1
 6node.data: true
 7node.master: true
 8# ----------------------------------- Paths ------------------------------------
 9# 日志目录和数据目录
10path.data: /usr/local/elasticsearch-6.3.2/data
11path.logs: /usr/local/elasticsearch-6.3.2/logs
12# ---------------------------------- Network -----------------------------------
13network.host: 192.168.231.60  # 本机ip的地址    
14http.port: 9200               # 端口为9200
15http.cors.enabled: true
16http.cors.allow-origin: "*"
17# --------------------------------- Discovery ----------------------------------
18discovery.zen.minimum_master_nodes: 1

2.0.3 安装Kibana

1[root@localhost ~]# tar -zxf kibana-6.3.2-linux-x86_64.tar.gz
2[root@localhost ~]# mv kibana-6.3.2 /usr/local/kibana-6.3.2
3[root@localhost ~]# cd /usr/local/kibana-6.3.2/config/
4[root@localhost ~]# vim kibana.yml

修改以下内容

 1# Kibana is served by a back end server. This setting specifies the port to use.
 2server.port: 15601              # Kibana的web页面端口
 3
 4# To allow connections from remote users, set this parameter to a non-loopback address.
 5server.host: "192.168.231.60"  # kibana本机ip地址
 6
 7# The URL of the Elasticsearch instance to use for all your queries.
 8elasticsearch.url: "http://192.168.231.60:9200" # 填写elasticsearch的ip地址
 9
10# dashboards. Kibana creates a new index if the index doesn't already exist.
11kibana.index: ".kibana" #  kibana的索引文件

2.0.4 安装Logstash

1[root@localhost ~]# tar -zxf logstash-6.3.2.tar.gz
2[root@localhost ~]# mv logstash-6.3.2  /usr/local/logstash-6.3.2
3[root@localhost ~]# cd /usr/local/logstash-6.3.2/config/
4[root@localhost config]# vim logstash.yml

修改以下内容

1path.config: /usr/local/logstash-6.3.2/config/*.conf # 此目录为logstash安装的目录
2config.reload.automatic: true # 配置文件自动加载
3config.reload.interval: 3s    # 每次加载间隔时间
4http.host: "192.168.231.60"   # 本机IP地址
5path.logs: /usr/local/logstash-6.3.2/logs  # 日志目录

2.0.5 安装filebeat

1[root@localhost ~]# tar -zxf filebeat-6.3.2-linux-x86_64.tar.gz
2[root@localhost ~]# tar -zxf filebeat-6.3.2 /usr/local/filebeat-6.3.2
3[root@localhost ~]# cd /usr/local/filebeat-6.3.2/
4[root@localhost ~]# vim filebeat.yml

修改以下内容

 1#  如果添加日志的话记得删除原来的
 2#=========================== Filebeat inputs =============================
 3filebeat:
 4  prospectors:
 5    - type: log    # 类型
 6      paths:
 7        - /var/log/nginx/access.log    #采取的日志目录
 8      tags: true
 9      
10# Change to true to enable this input configuration.
11  enabled: true
12
13#----------------------------- Logstash output --------------------------------
14output.logstash:
15        #  The Logstash hosts
16        hosts: ["192.168.231.60:5044"] # logstash默认端口5044
17
18
19#-------------------------- Elasticsearch output ------------------------------
20# output.elasticsearch:   //注释
21  # Array of hosts to connect to. //注释
22  #  hosts: ["localhost:9200"]  //注释

2.0.6 编写nginx-access.conf

1[root@localhost bin]# vim /usr/local/logstash-6.3.2/config/nginx-access.conf

nginx-access.conf

 1input {
 2 beats {
 3  port => 5044
 4   }
 5}
 6output {
 7 elasticsearch {
 8  hosts => "192.168.231.60:9200"
 9  index => "nginx-%{+YYYY.MM.dd}"
10   }
11}

2.0.7 启动服务

 1# 启动elasticsearch
 2[root@localhost ~]# cd /usr/local/elasticsearch-6.3.2/bin/
 3[root@localhost bin]# nohup ./elasticsearch &
 4
 5# 启动kibana
 6[root@localhost kibana-6.3.2]# cd /usr/local/kibana-6.3.2/bin/
 7[root@localhost bin]# nohup ./kibana &
 8
 9# 启动logstash
10[root@localhost bin]# cd /usr/local/logstash-6.3.2/bin/
11[root@localhost bin]# nohup ./logstash -t /usr/local/logstash-6.3.2/config/nginx-access.conf
12
13# 启动filebeat
14[root@localhost filebeat-6.3.2]# cd /usr/local/filebeat-6.3.2/
15[root@localhost filebeat-6.3.2]# nohup ./filebeat -c /usr/local/filebeat-6.3.2/filebeat.yml

2.0.8 验证结果

1访问192.168.231.60:15601
2出现kibana页面即可
