部署副本集

1.0 更改Mongo配置文件

1[root@localhost mongo]# vim conf/mongo.conf
2port=27017
3fork=true
4logpath=/usr/local/mongo/logs/mongodb.log
5logappend=true
6dbpath=/usr/local/mongo/data
7replSet=rs0 # 加入副本集名称,此名称要一致

启动Mongo

1[root@localhost mongo]# mongod -f conf/mongo.conf

1.1 主Mongo配置

 1[root@localhost mongo]# mongo
 2> conf=
 3    {
 4    "_id" : "rs0",
 5    "members" : [
 6        { "_id" : 0, "host" : "172.17.100.193:27017" },
 7        { "_id" : 1, "host" : "172.17.100.191:27017" }
 8        ]
 9    }
10# 填写mongo的ip地址即可
11> rs.initiate(conf) # 初始化副本集
12{ "ok" : 1 } # 如果出现ok1则表示成功

1.2 副本集更新

 1# 向副本集中添加成员
 2rs.add("172.17.100.191:27017")
 3 
 4# 从副本集中删除成员
 5rs.remove("172.17.100.191:27017")
 6 
 7# 向副本集中添加仲裁
 8rs.addArb("172.17.100.191:27017")
 9 
10# 向副本集中添加备份节点
11rs.add({"_id":3,"host":"172.17.100.191:27017","priority":0,"hidden":true})  // _id请依次递增

1.4 更新副本集优先级

1cfg = rs.conf()
2cfg.members[0].priority = 5  #  设置权重为5
3# members[0]里面填写你要更改的节点数字 id是1就写1 是2就设置2
4rs.reconfig(cfg) # 重新加载配置

1.5 查看副本集状态

1rs0:PRIMARY> rs.status()

1.6 查看副本集的配置信息

1rs0:PRIMARY> rs.conf()

1.7 查看节点复制信息

1rs0:PRIMARY> db.printSlaveReplicationInfo()

1.8 插入测试数据

1rs0:PRIMARY> for(var i=0;i<10000;i++){db.customer.insert({"name":"user"+i})}
2WriteResult({ "nInserted" : 1 })
3rs0:PRIMARY> db.customer.count()
410000
5#  在Secondary上查看客户数据是否已经同步：
6rs0:SECONDARY> rs.slaveOk()
7rs0:SECONDARY> db.customer.count()
810000

开启安全验证

先停止从上面的mongo然后在停止主上面的mongo

  • 在master上进行操作

2.0 创建用户

1rs0:PRIMARY> use admin
2rs0:PRIMARY> db.createUser({user: 'root', pwd: '123.com', roles: ['root']}) 
3#  创建root用户并且给予root权限,密码为123.com

1# 例如对某个库授权
2rs0:PRIMARY> use test
3rs0:PRIMARY> db.createUser({user:"admin",pwd:"admin",roles:[{role:"readWrite",db:"test"}]})
4# role角色具体可以看官网的分配权限
5# db: 指定授权的库

2.1 创建秘钥文件

1[root@localhost mongo]# openssl rand -base64 666 > /usr/local/mongo/keyfile

将生成的keyFile文件拷贝到其他节点服务器上，并修改文件的操作权限为 600

1# 更新启动配置文件,master上配置
2[root@localhost mongo]# vim conf/mongo.conf 
3auth=true
4oplogSize=100
5keyFile=/usr/local/mongo/keyfile

1# 更新启动配置文件,slave上配置
2[root@localhost mongo]# vim conf/mongo.conf 
3oplogSize=100
4keyFile=/usr/local/mongo/keyfile

2.2 启动副本集

1[root@localhost mongo]# mongod -f /usr/local/mongo/conf/mongo.conf
2# 先开master
3[root@localhost mongo]# mongod -f /usr/local/mongo/conf/mongo.conf 
4# 然后再从上开启服务,不加auth认证
