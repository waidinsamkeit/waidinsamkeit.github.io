  1. 你需要使用kubelet get node -o wide 查看你的集群状态信息

如上状态STATUS=Ready即可进行下面的操作

  1. 已经安装好Harbor并且和仓库已经集成Auth认证
  2. 主机之间全部相互免密完成
  3. gitee的公钥要和你的jenkins主机私钥匹配

１.１构建工具集

  • Jenkins

Jenkins是一款由Java编写的开源的持续集成工具。它运行在Servlet容器中（例如Apache Tomcat）。它支持软件配置管理（SCM）工具（包括AccuRev SCM、CVS、Subversion、Git、Perforce、Clearcase和RTC），可以执行基于Apache Ant和Apache Maven的项目，以及任意的Shell脚本和Windows批处理命令。Jenkins是一个有状态的应用，如果托管在K8S中，只运行一个副本，且需要持久化存储。

官网：https://jenkins.io/zh/

下载地址：https://jenkins.io/zh/download/

官方文档：https://jenkins.io/zh/doc/

  • Maven

Apache Maven，是一个软件（特别是Java软件）项目管理及自动构建工具，由Apache软件基金会所提供。Maven也可被用于构建和管理各种项目，例如C#，Ruby，Scala和其他语言编写的项目。Maven曾是Jakarta项目的子项目，现为由Apache软件基金会主持的独立Apache项目。在Devops流水线中，参与Java程序构建工具常用Maven。

官网：https://maven.apache.org/index.html

下载地址：https://archive.apache.org/dist/maven/

基础教程：https://www.runoob.com/maven/maven-setup.html

1.3 Jenkins持续交付到k8s上

一般都是公司自己使用持续集成,所以需要搭建私有仓库

此处没有根据大佬吧jenkins交付到k8s上,做一个简单的持续集成jenkins

再说一次,这个地方超级超级的难,因为要用docker来跑到k8s中

[root@kubernetes-master ~]# docker pull jenkins/jenkins:2.190.3 //我这直接用hub.docker.com的
[root@kubernetes-master jenkins]# docker tag jenkins/jenkins:2.190.3 192.168.1.177:8077/ci_cd/jenkins:v2.190.3 //打标签 这个192.168.1.177是你harbor仓库
[root@kubernetes-master jenkins]# docker push 192.168.1.177:8077/ci_cd/jenkins:v2.190.3 // 推送到harbor仓库
[root@kubernetes-master jenkins]# mkdir /usr/local/jenkins/
[root@kubernetes-master jenkins]# cd /usr/local/jenkins
[root@kubernetes-master jenkins]# cp /root/.ssh/id_rsa ./
[root@kubernetes-master jenkins]# cp /root/.docker/config.json
[root@kubernetes-master jenkins]# curl -fsSL get.docker.com -o get-docker.sh
[root@kubernetes-master jenkins]# chmod +x get-docker.sh
[root@kubernetes-master jenkins]# vim dockerfile
[root@kubernetes-master jenkins]# docker build . -t 192.168.1.177:8077/infra/jenkins:v2.190.3 //编辑完dockerfile在构建
[root@kubernetes-master jenkins]# docker push 192.168.1.177:8077/infra/jenkins:v2.190.3

１.４制作Dockerfile文件

FROM 192.168.1.177:8077/ci_cd/jenkins:v2.190.3
USER root
RUN /bin/cp /usr/share/zoneinfo/Asia/Shanghai /etc/localtime &&\
    echo 'Asia/Shanghai' > /etc/timezone
ADD id_rsa /root/.ssh/id_rsa
ADD config.json /root/.docker/config.json
ADD get-docker.sh /get-docker.sh
RUN echo "    StrictHostKeyChecking no" >> /etc/ssh/ssh_config &&\
    /get-docker.sh

这个dockerfile主要有以下几个用途

  • 设置容器用户为root
  • 设置容器内的时区为UTC
  • 将ssh私钥加入(使用gitee拉取代码需要匹配,你的公钥需要加入到gitee中)
  • 加入了登录自建harbor仓库的auth认证文件
  • 修改了ssh客户端的配置 连接一台新主机需要输入yes或no 设置no以后直接取消
  • 安装一个docker客户端 只是在jenkins里面做一个客户端

注意 由于jenkins属于私有镜像 为了安全你们要去你们的harbor上新建一个私有仓库

2.1Kubernetes创建jenkins命名空间

[root@kubernetes-master jenkins]# kubectl create ns infra

注意 此时我们需要把jenkins放入到infra这个单独的名称空间 由于jenkins放入到名称空间以后 从harbor上拉取镜像已经不仅仅需要的是docker login了。

你还需要做以下操作

[root@kubernetes-master jenkins]# kubectl create secret docker-registry harbor --docker-server=192.178.1.177:8077 --docker-username=admin --docker-password=Harbor12345 -n infra //需要为jenkins创建一个secret资源,类型为docker-registry 名字为harbor

2.2 Kubernetes中Pod共享数据

这里准备一个共享存储,采用NFS

[root@kubernetes-master jenkins]# yum -y install nfs-utils
  • 192.168.1.178作为共享存储的客户端,这里是重点一定是在node1节点上进行
  • 所有的Pod数据都需要挂载到共享存储
[root@kubernetes-node1 ~]# vim /etc/exports
/data/nfs-volume 192.168.1.0/24(rw,ro_root_squash) //共享给主机的IP段
[root@kubernetes-node1 ~]# mkdir /data/nfs-volume
[root@kubernetes-node1 ~]# mkdir /data/nfs-volume/jenkins_home
[root@kubernetes-node1 ~]# systemctl start nfs
[root@kubernetes-node1 ~]# systemctl enable nfs

准备jenkins资源配置清单

  • deployment
  • services
  • ingress
[root@kubernetes-master jenkins]# vim jenkins_dep.yaml
[root@kubernetes-master jenkins]# vim service.yaml
[root@kubernetes-master jenkins]# vim ingress.yaml

deployment原版

kind: Deployment
apiVersion: apps/v1 //注意 有些kubernetes版本会报错误 使用extensions/v1beta1即可 
metadata:
  name: jenkins
  namespace: infra
  labels: 
    name: jenkins
spec:
  replicas: 1
  selector:
    matchLabels: 
      name: jenkins
  template:
    metadata:
      labels: 
        app: jenkins 
        name: jenkins
    spec:
      volumes:
      - name: data
        nfs: 
          server: 192.168.1.178
          path: /data/nfs-volume/jenkins_home
      - name: docker
        hostPath: 
          path: /run/docker.sock
          type: ''
      containers:
      - name: jenkins
        image: 192.168.1.177:8077/infra/jenkins:v2.190.3
        ports:
        - containerPort: 8080
          protocol: TCP
        env:
        - name: JAVA_OPTS
          value: -Xmx512m -Xms512m
        volumeMounts:
        - name: data
          mountPath: /var/jenkins_home
        - name: docker
          mountPath: /run/docker.sock
      imagePullSecrets:
      - name: harbor
      securityContext: 
        runAsUser: 0
  strategy:
    type: RollingUpdate
    rollingUpdate: 
      maxUnavailable: 1
      maxSurge: 0
  revisionHistoryLimit: 7
  progressDeadlineSeconds: 600

deployment注释版

Kind: Deployment //Pod控制器类型 还有一种是daemonset,daemonset每个节点上都会运行一个
apiVersion: apps/v1 //注意 有些kubernetes版本会报错误 使用extensions/v1beta1即可
metadata:
  name: jenkins
  namespaces: infra
  labels:
    name: jenkins
spec:
  replicas: 1 //根据replicas启动 启动1个
  selector:
    matchLabels:
      name: jenkins
  template:   //Pod的模板
    metadate:
      lables:
        app: jenkins
        name: jenkins
    spec:
      volumes:
      - name: data
        nfs:
          server: kubernetes-node1  //这个地方写NFS服务的主机名或者IP
          path: /data/nfs-volume/jenkins_home //挂载目录
      - name: docker 
        hostpath:
          path: /run/docker.sock //挂载到/run/docker.sock 可以与docker服务段进行通信
        containers:
      - name: jenkins
        image: 192.168.1.177:8077/infra/jenkins:v2.190.3
        imagePullPolicy: IfNotPresent //分为 always never ifnotpresent
        ports:
        - containerPort: 8080 //往外映射端口为8080
          protocol: TCP
        env:
        - name: JAVA_OPTS
          value: -Xmx512m -Xms512m //赋予jenkins内存
        volumeMounts:
        - name: data
          mountPath: /var/jenkins_home //挂载内部jenkins的/var/jenkins_home到nfs的/data
        - name: docker
          mountPath: /run/docker.sock // 挂载内部/run/docker.sock 到docker存储
      imagePullSecrets: //镜像Secrets的资源类型 只要是私有仓库都需要create secrets
      - name: harbor    //名字为harbor 
      securityContext: 
        runAsUser: 0
  strategy:
    type: RollingUpdate
    rollingUpdate:  //默认采用滚动升级
      maxUnavailable: 1
      maxSurge: 0
  revisionHistoryLimit: 7 //留几分老数据
  progressDeadlineSeconds: 600 //到底容器运行多少时间依旧不正常 60s为1分钟

service

apiVersion: v1
metadata: 
  name: jenkins
  namespace: infra
spec:
  type: NodePort 
  ports: 
  - protocol: TCP
    port: 80
    targetPort: 8080
    nodePort: 30111

    app: jenkins

service注释版

kind: Service
apiVersion: v1
metadata: 
  name: jenkins
  namespace: infra
spec:
  type: NodePort //设置端口暴露节点的方式 NodePort和ingress
  ports:
  - protocol: TCP
    port: 80    //集群网络的80端口 
    targetPort: 8080 //容器内部运行端口
    nodePort: 30111 //供给外部映射访问的端口为Cluster IP+30111
  selector:
    app: jenkins

ingress 其实你也可以不用ingress 如果你没有dns解析的话

kind: Ingress
apiVersion: apps/v1
metadata: 
  name: jenkins

  rules:
  - host: jenkins.k.com
    http:
      paths:
      - path: /
        backend: 
          serviceName: jenkins
          servicePort: 80

2.4错误排查

[root@kubernetes-master jenkins]# kubectl describe  pods -n infra //可以查看Pod详细信息
[root@kubernetes-master jenkins]# kubectl get pods -n infra
NAME                       READY   STATUS    RESTARTS   AGE
jenkins-7bbd44dc8b-ngv8f   1/1     Running   0          4m17s
// 如果是Running就是运行正常

  1. 如果遇到拉取镜像错误

    Failed       7m36s                   kubelet, kubernetes-node1  Failed to pull image "192.168.1.177:8077/infra/jenkins:v2.190.3": rpc error: code = Unknown desc = Error response from daemon: Get https://192.168.1.177:8077/v2/: http: server gave HTTP response to HTTPS client

    解决方法1：cp /root/.docker/config.json到node节点上 因为harbor是通过认证密钥来获取镜像 这个错误大多数都是拉取不了镜像

    解决方法2：vim /etc/docker/daemon.json

    { “insecure-registries”:[“192.168.1.177:8077”] } 写入到daemon.json //设置不安全仓库

    systemctl daemon-reload

    systemctl restart docker
